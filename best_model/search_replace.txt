def masked_mean_np(X: np.ndarray, mask: np.ndarray) -> np.ndarray:
    cnt = np.clip(mask.sum(axis=1), 1.0, None)  # (U,1)
    return (X * mask).sum(axis=1) / cnt

def masked_std_np(X: np.ndarray, mask: np.ndarray, eps: float = 1e-6) -> np.ndarray:
    cnt = np.clip(mask.sum(axis=1), 1.0, None)  # (U,1)
    mean = (X * mask).sum(axis=1) / cnt
    var = ((X - mean[:, None, :]) ** 2 * mask).sum(axis=1) / cnt
    return np.sqrt(np.maximum(var, eps)).astype(np.float32)

def masked_trend_np(X: np.ndarray, mask: np.ndarray, eps: float = 1e-6) -> np.ndarray:
    """
    Per-feature linear trend over days (slope). Captures growth/decay.
    """
    U, T, D = X.shape
    t = np.arange(T, dtype=np.float32).reshape(1, T, 1)
    m = mask.astype(np.float32)

    cnt = np.clip(m.sum(axis=1), 1.0, None)  # (U,1)
    t_mean = (t * m).sum(axis=1) / cnt
    x_mean = (X * m).sum(axis=1) / cnt

    t_center = t - t_mean[:, None, :]
    x_center = X - x_mean[:, None, :]

    cov = (t_center * x_center * m).sum(axis=1) / cnt
    var_t = (t_center ** 2 * m).sum(axis=1) / cnt
    slope = cov / np.maximum(var_t, eps)
    return slope.astype(np.float32)
